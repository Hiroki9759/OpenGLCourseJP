#version 330

// CPUから受け取る頂点位置 (頂点バッファの中の1つ)
layout(location = 0) in vec3 in_position;

// どの頂点にも共通に使う変数
uniform mat4 u_mvpMat;

void main(void) {
    // 座標変換により、スクリーン上の位置を決定 (-1から1の範囲が描画される)
    gl_Position = u_mvpMat * vec4(in_position, 1.0);
}
